Page & Form Pattern

มาตรฐานการสร้างหน้าพร้อม FormGroup โดยใช้ Resolver-driven approach

ไฟล์ที่ต้องสร้าง

  • *.model.ts — Interface ของ model + FormData
  • *.form.ts — Class สำหรับสร้าง FormGroup
  • *.resolver.ts — Resolver โหลด form + data
  • *.service.ts — HTTP calls
  • *.component.ts/.html/.css — Component

หลักการ

  • ใช้ SicFromData<T> ครอบ FormGroup
  • สร้าง form ใน Resolver ก่อน component load
  • รับ form จาก route.snapshot.data
  • ใช้ formData.dirty สำหรับ unsaved guard

3. resolver.ts

// example.resolver.ts
import { inject } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { ActivatedRouteSnapshot, ResolveFn, Router } from '@angular/router';
import { catchError, EMPTY, map, tap } from 'rxjs';
import { SicFromData } from '../../../core/model/sic-from-data';
import { ExampleForm } from './example.form';
import { ExampleService } from './example.service';
import { ExampleFormData, ExampleModel } from './example.model';

// Create (ไม่โหลดข้อมูล)
export const exampleCreateResolver: ResolveFn<ExampleFormData> = () => {
  const fb = inject(FormBuilder);
  const form = ExampleForm.createForm(fb);
  return { example: new SicFromData<ExampleModel>(form) };
};

// Edit (โหลดข้อมูลแล้ว patchValue)
export const exampleEditResolver: ResolveFn<ExampleFormData> = (route) => {
  const fb = inject(FormBuilder);
  const service = inject(ExampleService);
  const router = inject(Router);
  const form = ExampleForm.createForm(fb);

  return service.getById(route.params['id']).pipe(
    tap((data: ExampleModel) => form.patchValue(data)),
    map(() => ({ example: new SicFromData<ExampleModel>(form) })),
    catchError(() => { router.navigate(['/not-found']); return EMPTY; })
  );
};

4. service.ts

// example.service.ts
import { inject, Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { environment } from '../../../../environments/environment';
import { ExampleModel } from './example.model';
import { Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class ExampleService {
  private readonly http = inject(HttpClient);
  private readonly base = environment.apiBaseUrl + '/api/ex/examples';

  getById(id: string): Observable<ExampleModel> {
    return this.http.get<ExampleModel>(`${this.base}/${id}`);
  }

  save(model: ExampleModel): Observable<string> {
    return this.http.post<string>(`${this.base}/save`, model);
  }
}

5. component.ts

// example.component.ts
@Component({ selector: 'app-example', standalone: true,
  imports: [CommonModule, ReactiveFormsModule, SicInput, SicButton, ...],
  templateUrl: './example.component.html'
})
export class ExampleComponent implements OnInit, CanComponentDeactivate {
  readonly route   = inject(ActivatedRoute);
  readonly router  = inject(Router);
  readonly service = inject(ExampleService);
  readonly dialog  = inject(DialogService);

  formData!: SicFromData<ExampleModel>;

  // ใช้สำหรับ CanDeactivateGuard
  pageDirty = () => this.formData.dirty;

  ngOnInit(): void {
    const data: ExampleFormData = this.route.snapshot.data['form'];
    this.formData = data.example;
  }

  onSubmit(): void {
    if (this.formData.invalid) {
      this.formData.formGroup.markAllAsTouched();
      return;
    }
    const model = this.formData.formGroup.getRawValue() as ExampleModel;
    this.service.save(model).subscribe({
      next: () => { this.dialog.alert('สำเร็จ', 'บันทึกเรียบร้อย'); this.router.navigate(['..']); },
      error: (err) => this.dialog.error('เกิดข้อผิดพลาด', err),
    });
  }
}

// app-translate-loader.ts (สรุปการทำงาน)
getTranslation(lang: string): Observable<TranslationObject> {
  // โหลด COMMON/ALL เสมอ
  const requests = [ this.loadModuleTranslations('COMMON', 'ALL', lang) ];

  // ถ้า context ไม่ใช่ COMMON/ALL ให้โหลดเพิ่ม
  if (this.moduleCode !== 'COMMON' || this.programCode !== 'ALL') {
    requests.push(this.loadModuleTranslations(this.moduleCode, this.programCode, lang));
  }

  // merge ผล (module เฉพาะ override COMMON ได้)
  return forkJoin(requests).pipe(map(results => deepMerge(...results)));
}

// ปรับ context สำหรับหน้าเฉพาะ (เรียกใน resolver หรือ component)

// Features/Ex/Example/GetExExamplePage.cs
public static class GetExExamplePage
{
    public class Query : Pageable, IRequest<PaginationBase<ExExample>>
    {
        public string? Keyword { get; set; }
    }

    public sealed class Handler(SicDbContext db) : IRequestHandler<Query, PaginationBase<ExExample>>
    {
        public async Task<PaginationBase<ExExample>> Handle(Query req, CancellationToken ct)
        {
            var page = req.PageNumber <= 0 ? 1 : req.PageNumber;
            var size = req.PageSize   <= 0 ? 10 : req.PageSize;

            IQueryable<ExExample> query = db.ExExamples.AsNoTracking();

            if (!string.IsNullOrWhiteSpace(req.Keyword))
                query = query.Where(x => x.MessageEn.Contains(req.Keyword));

            // ApplySorting จาก Extensions
            query = query.ApplySorting(req.Sorts, x => x.ExampleCode);

            var total = await query.LongCountAsync(ct);
            var data  = await query.Skip((page-1)*size).Take(size).ToArrayAsync(ct);

            return new PaginationBase<ExExample>
            {
                Data = data,
                Pageable = new Pageable { PageNumber=page, PageSize=size, TotalElements=total }
            };
        }
    }
}

// Features/Ex/Example/SaveExExample.cs
public static class SaveExExample
{
    // Command = DTO ที่รับจาก client
    public class Command : BaseModelState, IRequest<Guid?>, IMapFrom<ExExample>
    {
        public Guid? Id { get; set; }
        public string ExampleCode { get; set; } = default!;
        public string MessageEn   { get; set; } = default!;
        public string MessageLocal { get; set; } = default!;
        public DateTime StartDate { get; set; }
        public DateTime EndDate   { get; set; }
        public bool IsActive      { get; set; } = true;

        // AutoMapper mapping override
        public void Mapping(Profile profile)
        {
            profile.CreateMap<Command, ExExample>()
                .ForMember(d => d.RowVersion, o => o.Ignore());
        }
    }

    // Validator — FluentValidation
    public class Validator : AbstractValidator<Command>
    {
        public Validator()
        {
            RuleFor(x => x.State)
                .Must(s => s is EntityState.Added or EntityState.Modified)
                .WithMessage("State must be Added or Modified.");
            RuleFor(x => x.Id).NotEmpty()
                .When(x => x.State == EntityState.Modified)
                .WithMessage("Id is required when Modified.");
            RuleFor(x => x.RowVersion).NotNull()
                .When(x => x.State == EntityState.Modified);
            RuleFor(x => x.ExampleCode).NotEmpty().MaximumLength(50);
            RuleFor(x => x.MessageEn).NotEmpty();
            RuleFor(x => x.MessageLocal).NotEmpty();
            RuleFor(x => x.StartDate).NotEmpty();
        }
    }

    // Handler — business logic
    public sealed class Handler(SicDbContext db, IMapper mapper) : IRequestHandler<Command, Guid?>
    {
        public async Task<Guid?> Handle(Command request, CancellationToken ct)
        {
            var entity = mapper.Map<ExExample>(request);

            var item = entity.State switch
            {
                EntityState.Added    => entity,
                EntityState.Modified => await db.ExExamples
                    .FirstOrDefaultAsync(x => x.Id == entity.Id, ct),
                _                   => null
            };

            if (item is null) return null;

            if (entity.State == EntityState.Modified)
                mapper.Map(request, item);

            if (entity.State == EntityState.Added)
                db.ExExamples.Add(item);

            await db.SaveChangesAsync(ct);
            return item.Id;
        }
    }
}

public static class GridSaveExExample
{
    public class Command : BaseModelState, IRequest<Guid?>, IMapFrom<ExExample>
    {
        public Guid? Id          { get; set; }
        public string ExampleCode { get; set; } = default!;
        public string MessageEn   { get; set; } = default!;
        // ... fields
        public void Mapping(Profile p) =>
            p.CreateMap<Command, ExExample>().ForMember(d => d.RowVersion, o => o.Ignore());
    }

    // BatchCommand รับ array
    public class BatchCommand : IRequest<List<Guid?>>
    {
        public Command[] Items { get; set; } = [];
    }

    public class Validator : AbstractValidator<Command>
    {
        public Validator()
        {
            RuleFor(x => x.ExampleCode).NotEmpty().MaximumLength(50);
            // ...
        }
    }

    public sealed class Handler(SicDbContext db, IMapper mapper, IValidator<Command> validator)
        : IRequestHandler<BatchCommand, List<Guid?>>
    {
        public async Task<List<Guid?>> Handle(BatchCommand batch, CancellationToken ct)
        {
            var results = new List<Guid?>();
            foreach (var cmd in batch.Items)
            {
                await validator.ValidateAndThrowAsync(cmd, ct);
                var entity = mapper.Map<ExExample>(cmd);
                var item = cmd.State switch
                {
                    EntityState.Added    => entity,
                    EntityState.Modified => await db.ExExamples
                        .FirstOrDefaultAsync(x => x.Id == entity.Id, ct),
                    EntityState.Deleted  => await db.ExExamples
                        .FirstOrDefaultAsync(x => x.Id == entity.Id, ct),
                    _                   => null
                };
                if (item is null) { results.Add(null); continue; }
                if (cmd.State == EntityState.Modified) mapper.Map(cmd, item);
                if (cmd.State == EntityState.Added)    db.ExExamples.Add(item);
                if (cmd.State == EntityState.Deleted)  item.IsDelete = true;
                results.Add(item.Id);
            }
            await db.SaveChangesAsync(ct);
            return results;
        }
    }
}
}
